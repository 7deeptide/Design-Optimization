    \subsubsection*{Problem 1.}\label{problem-1.}

Minimize $f$ and find the optimum, $x^{*}$

\[ f(x,u) = \frac{1}{2} \left(\frac{x^2}{a^2} + \frac{u^2}{b^2} \right)\]

subject to the constraint

\[ g(x,u) = c -xu = 0.\]

Using the Lagrangian multipliers method we have

\begin{eqnarray*} H(x,u,\lambda) &=& f(x,u) + \lambda g(x,u)\\
                                               &=& \frac{1}{2} \left( \frac{x^2}{a^2} + \frac{u^2}{b^2} \right) + \lambda (c -xu). \end{eqnarray*}

The necessary conditions are
\begin{subequations}
	\label{eq1}
	\begin{align}
	\frac{\partial H}{\partial x} &= \frac{x}{a^2} - \lambda u = 0 \label{eq11}\\
\frac{\partial H}{\partial u} &= \frac{u}{b^2} - \lambda x = 0 \label{eq12}
	\end{align}
\end{subequations}

Solving equations \ref{eq11} and \ref{eq12} for $\lambda$ we have

\begin{equation} u = \frac{x}{\lambda a^2} \label{eq2} \end{equation}

and substituting equation \ref{eq2} into equation \ref{eq12} and simplifying we find the value of the Lagrangian multiplier

\[ \lambda = \frac{1}{ab} \]

which, we can use to find an expression for $x$,

\begin{equation} x = u \frac{a}{b}. \label{x} \end{equation}

To satisfy the sufficient conditions, and utilizing subscript notation for the derivatives, we have
\begin{subequations}
	\label{eq3}
	\begin{align}
	\left. \frac{\partial^2 f}{\partial x^2} \right|_{u=0} &= 0 \label{eq31}\\
\left. \frac{\partial^2 f}{\partial x^2} \right|_{u=0} &= H_{xx} - H_{xu}g_u^{-1}g_x - g_x^T(g_u^{-1})^T H_{ux} \label{eq32} \\
&+ g_x^T (g_u^{-1})^T H_{uu} g_u^{-1} g_x \nonumber
	\end{align}
\end{subequations}

Supplying values for the derivatives of $H$ and $g$ we have
\begin{eqnarray*}
\begin{split}
\frac{\partial^2 H}{\partial x^2} &= \frac{1}{a^2}\\
\frac{\partial^2 H}{\partial u^2} &= \frac{1}{b^2}\\
\frac{\partial^2 H}{\partial (xu)} &= \frac{\partial^2 H}{\partial (ux)} = - \lambda = -\frac{1}{ab}
\end{split}
\hspace{5 mm}
\begin{split}
\frac{\partial g}{\partial x} &= -u\\
\frac{\partial g}{\partial u} &= -x
\end{split}
\end{eqnarray*}

Substituting these values into equation \ref{eq32} we have

\begin{equation} \left. \frac{\partial^2 f}{\partial x^2} \right|_{u=0} = \frac{1}{a^2} + 2 \frac{u}{abx}	+ \frac{u^2}{b^2x^2} \end{equation}

which, substituting equation \ref{eq2} for $u$ and canceling terms we have

\[\left. \frac{\partial^2 f}{\partial x^2} \right|_{u=0} = \frac{4}{a^2} \]

This must be a positive number, thus the sufficient conditions are satisfied for a minimum. If we substitute equation \ref{x} into the constraint $g$ we find that the minimum is at
\[x^* = \sqrt{\frac{ac}{b}}, \hspace{6 mm} u^* = \sqrt{\frac{bc}{a}} \]
and
\[f^* = \frac{c}{ab} \]
 

    \subsubsection*{Problem 2.}\label{problem-2.}

Maximize V, where

\[ V(x,y,z) = 8xyz\]

subject to the constraint

\[\frac{x^2}{a^2} + \frac{y^2}{b^2} + \frac{z^2}{c^2}= 1.\]

We will use a direct substitution approach to turn this into an  unconstrained problem of two variables. First solving for $z$ in the constraint we have

\[z = c \sqrt{1- \frac{x^2}{a^2} - \frac{y^2}{b^2}}. \]

If we substitute this value for $z$ into the objective function, and note that we will minimize the negative to find the maximum we have

\[ \mbox{min} \hspace{2 mm} V(x,y) = -8xyc \sqrt{1- \frac{x^2}{a^2} - \frac{y^2}{b^2}}. \]

To satisfy the necessary conditions of the problem, and with some simplifications, we have

\begin{eqnarray*}
 \frac{\partial V}{\partial x} &=& -8yc \left(1 - \frac{x^2}{a^2} - \frac{y^2}{b^2} \right)^{-1/2}\left( 1 - \frac{2x^2}{a^2} - \frac{y^2}{b^2} \right)= 0 \\
 \frac{\partial V}{\partial y} &=& -8xc \left(1 - \frac{x^2}{a^2} - \frac{y^2}{b^2} \right)^{-1/2}\left( 1 - \frac{x^2}{a^2} - \frac{2y^2}{b^2} \right)= 0
\end{eqnarray*}

which we can simplify as 

\begin{eqnarray*}
 1 - \frac{2x^2}{a^2} - \frac{y^2}{b^2} = 0 \\
 1 - \frac{x^2}{a^2} - \frac{2y^2}{b^2} = 0.
\end{eqnarray*}

Substituting one into the other we have 
\[ x^* = \frac{a}{\sqrt{3}}, \hspace{3 mm} y^* = \frac{b}{\sqrt{3}}, \]

and by extension

\[ z^* = \frac{c}{\sqrt{3}}, \hspace{3 mm} \mbox{and} \hspace{3 mm} V^* = \frac{8a^2b^2c^2}{3 \sqrt{3}}.\]

To satisfy the sufficient condition for a minimum we have

\begin{eqnarray*}
\left. \frac{\partial^2 V}{\partial x^2} \right|_{x=x^*,y=y^*} &=& \frac{8 c x y (3 a^2 b^2-2 b^2 x^2-3 a^2 y^2)}{a^4 b^2 \left(1-\frac{x^2}{a^2}-\frac{y^2}{b^2} \right)^{3/2}} = \frac{-32cb}{a}\\
\left. \frac{\partial^2 V}{\partial y^2} \right|_{x=x^*,y=y^*} &=& \frac{-32cb}{a} \\
\left. \frac{\partial^2 V}{\partial x \partial y} \right|_{x=x^*,y=y^*} &=& \frac{-8 c (a^4 b^4-3 a^4 b^2 y^2+2 a^4 y^4-3 a^2 b^4 x^2+3 a^2 b^2 x^2 y^2+2 b^4 x^4)}{a^4 b^4 \left( \frac{-x^2}{a^2}-\frac{y^2}{b^2}+1 \right)^{3/2}} \\
&=& \frac{112c}{27}\left(\frac{1}{3} \right)^{-3/2}
\end{eqnarray*} 

If we assume that the coefficients $a$, $b$, $c$ are positive and real then 

\[\frac{\partial^2 V}{\partial x^2} < 0\]

and given the sufficient condition for a minimum that the determinant of the Hessian matrix

 \[ \frac{\partial^2 V}{\partial x^2} \frac{\partial^2 V}{\partial y^2} - \left(\frac{\partial^2 V}{\partial x \partial y)} \right)^2 > 0 \]
 
then we have a maximum (or the minimum of the negative of the objective function) when

\[ \frac{b^2}{a^2} > \frac{9}{32^2} \left(\frac{112}{27} \right)^2 \approx 0.151\]
